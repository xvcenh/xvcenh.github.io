// VMC 正运动学解算
void VMC_calc_1(vmc_leg_t *vmc, float dt) {
  // 1. 计算B、D点坐标
  vmc->YD = l4 * sin(phi4);
  vmc->YB = l1 * sin(phi1);
  vmc->XD = l5 + l4 * cos(phi4);
  vmc->XB = l1 * cos(phi1);

  // 2. 计算BD距离
  lBD = sqrt((XD-XB)² + (YD-YB)²);

  // 3. 计算phi2 (闭链约束)
  A0 = 2*l2*(XD-XB);
  B0 = 2*l2*(YD-YB);
  C0 = l2² + lBD² - l3²;
  phi2 = 2*atan2(B0+sqrt(A0²+B0²-C0²), A0+C0);

  // 4. 计算C点坐标
  XC = l1*cos(phi1) + l2*cos(phi2);
  YC = l1*sin(phi1) + l2*sin(phi2);

  // 5. 计算腿长和腿角度
  L0 = sqrt((XC-l5/2)² + YC²);
  phi0 = atan2(YC, XC-l5/2);
  theta = π/2 - Pitch - phi0;
}

// K值随腿长L0变化 (增益调度)
for(i=0; i<12; i++) {
  K[i] = c[i][0]*L0³
       + c[i][1]*L0²
       + c[i][2]*L0
       + c[i][3];
}

// 控制律
wheel_T = K[0]*theta + K[1]*d_theta
        + K[2]*x_err + K[3]*v_err
        + K[4]*pitch + K[5]*d_pitch;

Tp = K[6]*theta + K[7]*d_theta
   + K[8]*x_err + K[9]*v_err
   + K[10]*pitch + K[11]*d_pitch;

// 离地检测算法
uint8_t ground_detection(vmc_leg_t *vmc, INS_t *ins) {
  // 估算地面法向力 (牛顿第二定律)
  vmc->FN = F0*cos(theta)
          + Tp*sin(theta)/L0
          + 0.6*(ins->MotionAccel_z
          - dd_L0*cos(theta)
          + 2*d_L0*d_theta*sin(theta)
          + L0*dd_theta*sin(theta)
          + L0*d_theta²*cos(theta));

  // 滑动平均滤波
  aver[0] = aver[1];
  aver[1] = aver[2];
  aver[2] = aver[3];
  aver[3] = FN;
  avg = 0.25*(aver[0]+aver[1]+aver[2]+aver[3]);

  // 判断离地
  if(avg < 3.0f) return 1; // 离地
  else return 0;           // 着地
}
