📖 算法说明

VMC将五连杆机构映射为虚拟腿(L₀, θ)。A、E为固定点，B、D由关节电机驱动。 C为末端执行器，通过闭链约束求解。橙色区域为C点可达工作空间。

// VMC 正运动学 - 从关节角度计算腿长和腿角度
void VMC_calc_1(vmc_leg_t *vmc, float dt) {
  // 1. 计算B、D点坐标
  vmc->YD = l4 * sinf(phi4);
  vmc->YB = l1 * sinf(phi1);
  vmc->XD = l5 + l4 * cosf(phi4);
  vmc->XB = l1 * cosf(phi1);

  // 2. BD距离 + 闭链约束求解phi2
  lBD = sqrtf((XD-XB)² + (YD-YB)²);
  phi2 = 2*atan2f(B0+sqrtf(A0²+B0²-C0²), A0+C0);

  // 3. C点坐标
  XC = l1*cosf(phi1) + l2*cosf(phi2);
  YC = l1*sinf(phi1) + l2*sinf(phi2);

  // 4. 腿长和腿角度
  L0 = sqrtf((XC-l5/2)² + YC²);
  phi0 = atan2f(YC, XC-l5/2);
  theta = π/2 - Pitch - phi0;
}
// LQR 状态反馈控制律
// K值随腿长L0变化 (增益调度)
for(i=0; i<12; i++)
  K[i] = Poly[i][0]*L0³ + Poly[i][1]*L0² + Poly[i][2]*L0 + Poly[i][3];

// 轮毂电机力矩
wheel_T = K[0]*theta + K[1]*d_theta + K[2]*x_err
        + K[3]*v_err + K[4]*pitch  + K[5]*d_pitch;

// 髋关节力矩
Tp = K[6]*theta + K[7]*d_theta + K[8]*x_err
   + K[9]*v_err + K[10]*pitch + K[11]*d_pitch;
// 离地检测 - 牛顿第二定律估算地面反力
FN = F0*cosf(theta) + Tp*sinf(theta)/L0
   + 0.6*(Accel_z - dd_L0*cosf(theta) + 2*d_L0*d_theta*sinf(theta) ...);

avg = 0.25*(buf[0]+buf[1]+buf[2]+FN);  // 滑动平均
if(avg < 3.0f) return 1; // 离地
// 跳跃状态机
case 1: // 压缩蓄能
  F0 = mg/cosf(theta) + PID(L0, 0.08);
  if(L0 < 0.10 && timer++ > 10) → 进入上升
case 2: // 爆发伸展
  F0 = mg/cosf(theta) + PID(L0, 0.40);
  if(L0 > 0.18 && timer++ > 2) → 进入缩腿
case 3: // 落地缓冲
  F0 = PID(L0, 0.10);
  if(L0 < 0.15 && timer++ > 3) → 回到站立
