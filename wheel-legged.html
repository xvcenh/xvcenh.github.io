📖 算法说明

VMC (Virtual Model Control) 通过五连杆机构的正运动学，将两个关节电机角度 (φ₁, φ₄) 映射为虚拟腿长 L₀ 和腿角度 θ。再通过雅可比矩阵将期望的力/力矩 (F₀, Tp) 映射回关节电机力矩。

// VMC 正运动学 - 从关节角度计算腿长和腿角度
void VMC_calc_1(vmc_leg_t *vmc, float dt) {
  // 1. 计算B、D点坐标
  vmc->YD = l4 * sinf(phi4);     // D点y坐标
  vmc->YB = l1 * sinf(phi1);     // B点y坐标
  vmc->XD = l5 + l4 * cosf(phi4); // D点x坐标
  vmc->XB = l1 * cosf(phi1);     // B点x坐标

  // 2. BD距离
  lBD = sqrtf((XD-XB)² + (YD-YB)²);

  // 3. 闭链约束求解phi2
  A0 = 2*l2*(XD-XB);
  B0 = 2*l2*(YD-YB);
  C0 = l2² + lBD² - l3²;
  phi2 = 2*atan2f(B0+sqrtf(A0²+B0²-C0²), A0+C0);

  // 4. C点坐标 (末端执行器)
  XC = l1*cosf(phi1) + l2*cosf(phi2);
  YC = l1*sinf(phi1) + l2*sinf(phi2);

  // 5. 极坐标: 腿长L0, 腿角度phi0
  L0 = sqrtf((XC-l5/2)² + YC²);
  phi0 = atan2f(YC, XC-l5/2);

  // 6. theta给LQR用
  theta = π/2 - Pitch - phi0;
  d_theta = -Gyro_Y - d_phi0;
}

for(i=0; i<12; i++) {

// 轮毂电机力矩输出

        + K[2]*x_err  + K[3]*v_err
        + K[4]*pitch  + K[5]*d_pitch;

// 髋关节力矩输出
Tp = K[6]*theta  + K[7]*d_theta
   + K[8]*x_err  + K[9]*v_err
   + K[10]*pitch + K[11]*d_pitch;

torque[0] = j11*F0 + j12*Tp;  // 前关节
torque[1] = j21*F0 + j22*Tp;  // 后关节
