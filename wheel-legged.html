// VMC 正运动学
void VMC_calc_1(vmc_leg_t *vmc, float dt) {
  vmc->YD = l4 * sinf(phi4);
  vmc->YB = l1 * sinf(phi1);
  vmc->XD = l5 + l4 * cosf(phi4);
  vmc->XB = l1 * cosf(phi1);
  lBD = sqrtf((XD-XB)² + (YD-YB)²);
  phi2 = 2*atan2f(B0+sqrtf(A0²+B0²-C0²), A0+C0);
  XC = l1*cosf(phi1) + l2*cosf(phi2);
  YC = l1*sinf(phi1) + l2*sinf(phi2);
  L0 = sqrtf((XC-l5/2)² + YC²);
  phi0 = atan2f(YC, XC-l5/2);
  theta = π/2 - Pitch - phi0;
}
// LQR 状态反馈控制律
for(i=0; i<12; i++)
  K[i] = Poly[i][0]*L0³ + Poly[i][1]*L0² + Poly[i][2]*L0 + Poly[i][3];
wheel_T = K[0]*theta + K[1]*d_theta + K[2]*x_err + K[3]*v_err + K[4]*pitch + K[5]*d_pitch;
Tp = K[6]*theta + K[7]*d_theta + K[8]*x_err + K[9]*v_err + K[10]*pitch + K[11]*d_pitch;
// 离地检测
FN = F0*cosf(theta) + Tp*sinf(theta)/L0 + 0.6*(Accel_z - dd_L0*cosf(theta) ...);
avg = 0.25*(buf[0]+buf[1]+buf[2]+FN);
